  • x %>% f is equivalent to f(x)

  • x %>% f(y) is equivalent to f(x, y)

  • x %>% f(y, .) is equivalent to f(y, x)

plot_yearly_means <- function(data, metals, genuses, sites){
  
  loctable <- suppressMessages(read_csv("loctable.csv"))
  
  agg_data <- filter(data, metal %in% metals, genus %in% genuses, loc %in% sites) %>% 
    left_join(locations, by = "loc") %>% 
    group_by(year, genus, name, metal, lab) %>% 
    summarise(value = mean(value), .groups = "drop")
  
  p <- agg_data %>%   
    ggplot(aes(x = year, y = value)) +
    geom_smooth(color = "black", size = .5, method = "loess", formula = y ~ x) +
    geom_point(color = "steelblue") + 
    theme_bw() +
    scale_y_continuous(expand = expansion(mult = c(0, 0)), limits = c(0, NA)) +
    xlab("") + ylab("Concentration (ng/g)") +
    facet_grid(~name)
  
  p
}
